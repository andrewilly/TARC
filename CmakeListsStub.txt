cmake_minimum_required(VERSION 3.15)
project(TARC_SFX_Stub LANGUAGES CXX)

# Forza C++17
set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

# Trova i pacchetti installati tramite vcpkg
find_package(LibLZMA REQUIRED)
find_package(zstd CONFIG REQUIRED)
find_package(xxHash CONFIG REQUIRED)

# Sorgenti necessari (percorsi relativi dalla cartella build_stub)
set(STUB_SOURCES
    ../src/stub.cpp
    ../src/engine.cpp
    ../src/io.cpp
    ../src/ui.cpp
)

add_executable(tarc_sfx_stub ${STUB_SOURCES})

# Definizioni per Windows
target_compile_definitions(tarc_sfx_stub PRIVATE _CRT_SECURE_NO_WARNINGS WIN32_LEAN_AND_MEAN)

# Directory di inclusione
target_include_directories(tarc_sfx_stub PRIVATE 
    ../src 
    ../include
    ${LIBLZMA_INCLUDE_DIRS}
)

# LINK DELLE LIBRERIE: Usiamo i nomi dei target ufficiali di vcpkg
target_link_libraries(tarc_sfx_stub PRIVATE 
    LibLZMA::LibLZMA
    zstd::libzstd_static
    xxHash::xxhash
)

set_target_properties(tarc_sfx_stub PROPERTIES OUTPUT_NAME "tarc_sfx_stub")
