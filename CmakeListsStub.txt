cmake_minimum_required(VERSION 3.15)
project(TARC_SFX_Stub LANGUAGES CXX)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

# Forza il link statico delle librerie C/C++ (MSVC)
set(CMAKE_MSVC_RUNTIME_LIBRARY "MultiThreaded$<$<CONFIG:Debug>:Debug>")
# Forza vcpkg a usare il triplet statico
set(VCPKG_TARGET_TRIPLET "x64-windows-static")

# 1. Trova i pacchetti (vcpkg li rende disponibili qui)
find_package(LibLZMA REQUIRED)
find_package(zstd CONFIG REQUIRED)
find_package(xxHash CONFIG REQUIRED)

set(STUB_SOURCES
    ../src/stub.cpp
    ../src/engine.cpp
    ../src/io.cpp
    ../src/ui.cpp
)

add_executable(tarc_sfx_stub ${STUB_SOURCES})

target_compile_definitions(tarc_sfx_stub PRIVATE _CRT_SECURE_NO_WARNINGS WIN32_LEAN_AND_MEAN)

target_include_directories(tarc_sfx_stub PRIVATE 
    ../src 
    ../include
)

# 2. LINK DELLE LIBRERIE
# Usiamo i nomi esatti che vcpkg ha confermato nei tuoi log:
target_link_libraries(tarc_sfx_stub PRIVATE 
    LibLZMA::LibLZMA
    zstd::libzstd
    xxHash::xxhash
)

set_target_properties(tarc_sfx_stub PROPERTIES OUTPUT_NAME "tarc_sfx_stub")
