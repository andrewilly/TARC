cmake_minimum_required(VERSION 3.15)
project(TARC_SFX_Stub LANGUAGES CXX)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

# Forza l'uso del runtime statico (importante per non richiedere DLL di Visual C++)
set(CMAKE_MSVC_RUNTIME_LIBRARY "MultiThreaded$<$<CONFIG:Debug>:Debug>")

# 1. Trova i pacchetti (Approccio compatibile con vcpkg static)
find_package(LibLZMA REQUIRED)
find_package(zstd REQUIRED) # Rimosso CONFIG se fallisce
find_package(xxHash REQUIRED)

set(STUB_SOURCES
    ../src/stub.cpp
    ../src/engine.cpp
    ../src/io.cpp
    ../src/ui.cpp
)

add_executable(tarc_sfx_stub ${STUB_SOURCES})

target_compile_definitions(tarc_sfx_stub PRIVATE _CRT_SECURE_NO_WARNINGS WIN32_LEAN_AND_MEAN)

target_include_directories(tarc_sfx_stub PRIVATE ../src ../include)

# 2. LINK DELLE LIBRERIE
# Utilizziamo i target standard esportati da vcpkg
target_link_libraries(tarc_sfx_stub PRIVATE 
    LibLZMA::LibLZMA
    $<IF:$<TARGET_EXISTS:zstd::libzstd_static>,zstd::libzstd_static,zstd::libzstd>
    xxHash::xxhash
)

set_target_properties(tarc_sfx_stub PROPERTIES OUTPUT_NAME "tarc_sfx_stub")
